
  核心概念

  生成器就是"按需生成值，而不是一次性生成所有值"。

  # 普通函数：一次性生成所有值，占内存
  def get_squares(n):
      return [x**2 for x in range(n)]   # 列表，全在内存里

  # 生成器：一个一个吐，不占内存
  def get_squares_gen(n):
      for x in range(n):
          yield x**2                     # yield = 暂停并返回值，下次继续

  对比

  # 普通列表
  lst = [x**2 for x in range(10_000_000)]   # 内存直接爆炸警告
  sum(lst)

  # 生成器 ✔
  gen = (x**2 for x in range(10_000_000))    # 不占内存，用的时候才算
  sum(gen)                                    # 边算边加

  最典型场景：读大文件

  # 错误写法：一次读全部
  lines = open("large_file.txt").readlines()  # 文件太大内存崩

  # 正确写法：用生成器逐行读 ✔
      process(line)                            # 一次只处理一行
